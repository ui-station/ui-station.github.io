import numpy as np

def count_water_pixels(data,lake_id):
    """

        count_water = np.count_nonzero(scl_image == 6)  # ë¬¼

                    scl_image == 5) | (scl_image == 6) | (scl_image == 7) | (scl_image == 8) | (scl_image == 9) | (
